poupanca_mensal = salario * 0.15  # 15% de 2000 = 300
acumulado = 0
mes = 0

while acumulado < meta:
    mes += 1
    
    # 1. Guarda o valor mensal 
    acumulado += poupanca_mensal
    
    # 2. Despesa fixa a cada 5 meses
    if mes % 5 == 0:
        acumulado -= 300
        
    # 3. Bônus a cada 8 meses 
    if mes % 8 == 0:
        acumulado += 800

sobra = acumulado - meta

def buscar_caminho(G, origem, destino):
    fila = [origem]
    visitados = []
 
    while len(fila) > 0:
        atual = fila.pop(0)
        if atual == destino:
            return True
        if atual not in visitados:
            visitados.append(atual) 
            vizinhos = G[atual]
            for v in vizinhos:
                if v not in visitados:
                    fila.append(v) 
                    
    return False

def BemFormada(s):
  p = []
  for c in s:
    if c == ')':  #menina fofa
      if p[-1] == '(': p.pop()  #se fofo match
      else: return False  #deu ruim sai
    else:
      p.append(c) #menino fofo vai para a pilha de espera
  return len(p) == 0  #se sobrou menino também deu ruim
