def string_implosion(s):
   k = len(s)
   resp = ""
   while k >= 0:
      resp = resp + s[:k]
      k = k - 1
   return resp

def sanduiche_de_consoantes(s):
    vogais = 'aeiou'
    cont = 0
    for i in range(len(s) - 2):
        if s[i].lower() not in vogais and s[i+2].lower() not in vogais:
            cont += 1
    return cont

def soma_impares_finais(n):
   n = str(n)[::-1]
   soma = 0
   for c in n:
      if int(c) % 2 != 0:
         soma += int(c)
      else:
         break
   return soma
