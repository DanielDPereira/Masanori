def max_repeticoes(palavra, letra):
   lista = []
   cont = 0
   for letra2 in palavra:
      if letra == letra2:
         cont += 1
      else:
         lista.append(cont)
         cont = 0
   return max(lista)

def string_implosion(s):
   k = len(s)
   resp = ""
   while k > 0:
      resp = resp + s[:k]
      k = k - 1
   return resp
