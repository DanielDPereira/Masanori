def sanduiche_de_consoantes(s):
   vogais = 'aeiou'
   res = 0
   t = len(s) - 2
   for a in s[:t]:
      for b in s[2:]:
         if a not in vogais and b not in vogais:
            res += 1
   return res

A função retorna true se n for igual a soma de um par de números distintos da lista.
def diferença_na_lista(n, lista):
   return n in [x - y for x in lista for y in lista if x != y]

A função retorna a quantidade de zeros no final de n.
def soma_impares_finais(n):
   n = str(n)[::-1]
   lista = []
   for c in n:
      if int(c) % 2 != 0:
         lista.append(int(c))
      else:
         break
   return sum(lista)

A função possui outra função que retorna o último caractere de x, que depois é usado como parâmetro no key= para o sorted ordenar as palavras pela última letra.
def ordena_por_tamanho(nomes):
   return sorted(nomes, key=len)

def inverte(frase):
   lista = frase.split()
   lista2 = []
   for p in lista:
      lista2.append(p[::-1])
   return ' '.join(lista2)
